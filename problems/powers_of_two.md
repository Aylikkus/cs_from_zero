```csharp
int[] powersOfTwo(int n)
{
    /*
     * Решение
    */
}

powersOfTwo(0); // => [1] (2^0)
powersOfTwo(2); // => [1, 2, 4] (2^0, 2^1, 2^2)
powersOfTwo(3); // => [1, 2, 4, 8] (2^0, 2^1, 2^2, 2^3)
powersOfTwo(-4); // => [-1]
